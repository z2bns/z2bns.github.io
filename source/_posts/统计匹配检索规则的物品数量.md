class Solution {
public:
    int countMatches(vector<vector<string>>& items, string ruleKey, string ruleValue) {
        int index=0,result=0;
        if(ruleKey=="color"){
            index++;
        }else if(ruleKey=="name"){
            index+=2;
        }
        int m=items.size();
        for(int i=0;i<m;i++){
            if(items[i][index]==ruleValue){
                result++;
            }
        }
        return result;
    }
};
